## 三、OneWire.c 驱动文件编写

### 步骤1：创建文件
**文件路径**: `Hardware/OneWire.c`

### 步骤2：写入内容

```c
/*
	# DS18B20单总线温度传感器驱动代码
	1. 本驱动配合蓝桥杯CT107D开发板使用
	2. 底层通信函数（Delay_OneWire/Write_DS18B20/Read_DS18B20/init_ds18b20）
	   由赛题BSP提供，放在本文件开头，严禁修改！
	3. 本文件在官方底层函数基础上封装高级温度读取函数
*/

#include <STC15F2K60S2.H>
#include "intrins.h"
#include "Public.h"         // 包含u8/u16/u32等统一类型定义
#include "OneWire.h"        // 自身头文件

/* ==================== DQ引脚定义（P1.4） ==================== */
sbit DQ = P1^4;

/* ==================== 官方底层函数（与官方代码完全一致，严禁修改） ==================== */

/**
 * @brief 单总线延时函数
 * @param t 延时单位数
 * @details 每单位延时约14us（11.0592MHz晶振下）
 * @note 官方函数名：Delay_OneWire
 */
void Delay_OneWire(unsigned int t)
{
	unsigned char i;
	while(t--){
		for(i=0;i<12;i++);
	}
}

/**
 * @brief 向DS18B20写入一个字节
 * @param dat 要写入的8位数据
 * @details 低位先发(LSB)，每位写入需要一个写时隙
 * @note 官方函数名：Write_DS18B20
 */
void Write_DS18B20(unsigned char dat)
{
	unsigned char i;
	for(i=0;i<8;i++)
	{
		DQ = 0;
		DQ = dat&0x01;
		Delay_OneWire(5);
		DQ = 1;
		dat >>= 1;
	}
	Delay_OneWire(5);
}

/**
 * @brief 从DS18B20读取一个字节
 * @retval 读取到的8位数据
 * @details 低位先收(LSB)，每位读取需要一个读时隙
 * @note 官方函数名：Read_DS18B20
 */
unsigned char Read_DS18B20(void)
{
	unsigned char i;
	unsigned char dat;
  
	for(i=0;i<8;i++)
	{
		DQ = 0;
		dat >>= 1;
		DQ = 1;
		if(DQ)
		{
			dat |= 0x80;
		}	    
		Delay_OneWire(5);
	}
	return dat;
}

/**
 * @brief DS18B20初始化（复位脉冲）
 * @retval 0: 初始化失败（无响应），1: 初始化成功
 * @details 主机发送复位脉冲，等待DS18B20应答脉冲
 * @note 官方函数名：init_ds18b20
 */
bit init_ds18b20(void)
{
  	bit initflag = 0;
  	
  	DQ = 1;
  	Delay_OneWire(12);
  	DQ = 0;
  	Delay_OneWire(80);
  	DQ = 1;
  	Delay_OneWire(10); 
    initflag = DQ;     
  	Delay_OneWire(5);
  
  	return initflag;
}

/* ==================== 扩展高级函数（基于官方底层封装） ==================== */

/**
 * @brief 读取DS18B20原始温度数据（16位有符号）
 * @retval 16位原始温度数据
 * @details 启动转换并读取温度，合并为一个函数
 * @note 实际温度 = 原始数据 / 16.0
 */
s16 DS18B20_ReadRawTemp(void)
{
    unsigned char low, high;
    s16 temp;
    
    // 启动转换
    init_ds18b20();
    Write_DS18B20(0xCC);      // 跳过ROM命令
    Write_DS18B20(0x44);      // 启动温度转换命令
    Delay_OneWire(200);       // 等待转换（12位需要约750ms，实际200单位约2.8ms，赛题通常够用）
    
    // 读取温度
    init_ds18b20();
    Write_DS18B20(0xCC);      // 跳过ROM命令
    Write_DS18B20(0xBE);      // 读取温度寄存器命令
    
    low = Read_DS18B20();      // 先读低字节
    high = Read_DS18B20();     // 再读高字节
    
    temp = (high << 8) | low;  // 组合为16位有符号数
    return temp;
}

/**
 * @brief 首次读取温度（用于上电初始化，避免显示85）
 * @details 上电后先读一次温度，丢弃85的初始值
 */
void DS18B20_InitRead(void)
{
    DS18B20_ReadRawTemp();  // 上电首次读取，丢弃85
}

/**
 * @brief 读取DS18B20温度值（可变小数位）
 * @param Decimal_Place 小数位数：0=整数, 1=1位小数, 2=2位小数
 * @retval 温度值放大相应倍数
 * @details 通用函数，根据参数返回不同精度
 */
s16 DS18B20_ReadTemp_X(u8 Decimal_Place)
{
	s16 raw = DS18B20_ReadRawTemp();
	switch(Decimal_Place)
	{
		case 0: return raw / 16;              // 整数
		case 1: return (raw * 10) / 16;       // 1位小数（如255=25.5）
		case 2: return (raw / 4) * 25;          // 2位小数（如2506=25.06）
		default: return raw / 16;
	}
}

/**
 * @brief 读取DS18B20温度值（float版本）
 * @retval 实际温度值，单位°C，例如25.5表示25.5°C
 * @note 使用浮点运算，消耗较多资源，仅用于需要精确浮点计算的场景
 */
float rd_temperature(void)
{
    s16 raw = DS18B20_ReadRawTemp();
    return raw / 16.0f;
}
```

## 四、User/main.c 中调用示例

```c
#include "System/Init.h"        // 系统初始化
#include "Hardware/OneWire.h"     // DS18B20驱动头文件
#include "Hardware/Nixie.h"     // 数码管驱动（用于显示温度）

void main(void)
{
    // 1. 系统初始化
    System_Init();
    
    // 2. DS18B20上电初始化（必须！否则显示85）
    // 上电首次读取丢弃85的初始值
    DS18B20_InitRead();
    
    // 3. 主循环：循环读取并显示温度
    while(1)
    {
        s16 temp_int;    // 整数温度
        s16 temp_x10;    // 带1位小数温度
        
        // 读取整数温度（如25°C）
        temp_int = DS18B20_ReadTemp_X(0);
        
        // 或读取带1位小数温度（如255 = 25.5°C）
        temp_x10 = DS18B20_ReadTemp_X(1);
        
        // 显示温度（示例：显示整数温度）
        Nixie_DisplayTemp(temp_int);
    }
}
```

## 五、关键要点速查

| 要点 | 说明 |
|------|------|
| **ROM命令** | 0xCC=Skip ROM（单设备），0x44=Convert T（启动转换），0xBE=Read Scratchpad（读取） |
| **分辨率** | 默认12位，最小单位0.0625°C，转换时间约750ms |
| **上电85问题** | 首次必须执行DS18B20_InitRead()丢弃85 |
| **温度计算** | 原始数据 / 16 = 实际温度（12位分辨率） |
| **整数运算** | 建议使用DS18B20_ReadTemp_X()避免浮点 |
| **DQ引脚** | CT107D开发板P1.4，已外接上拉电阻 |

## 六、常见问题

**Q1: 显示85°C？**
- 原因：上电默认值，未执行首次读取
- 解决：在main函数初始化时调用DS18B20_InitRead()

**Q2: 读取的温度跳变/不稳定？**
- 原因：转换时间不够，延时太短
- 解决：增加Delay_OneWire(200)或更多

**Q3: 编译报错找不到Delay_OneWire？**
- 原因：未包含OneWire.h或文件未添加到工程
- 解决：确认文件路径正确，已添加到Keil工程
