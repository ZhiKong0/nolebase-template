## 二、DS1302.h 头文件编写

### 步骤2：写入内容

#ifndef __DS1302_H__
#define __DS1302_H__

#include "STC15F2K60S2.H"   // 单片机寄存器定义
#include "Public.h"         // u8/u16/u32 类型定义（在System文件夹）

//============================
// 外部变量声明（供其他文件使用）
//============================

/**
 * DS1302时间寄存器写地址数组（秒、分、时、日、月、星期、年）
 * 写地址为偶数：0x80, 0x82, 0x84, 0x86, 0x88, 0x8A, 0x8C
 */
extern code u8 DS1302_Write_adr[7];

/**
 * DS1302时间寄存器读地址数组
 * 读地址为奇数：0x81, 0x83, 0x85, 0x87, 0x89, 0x8B, 0x8D
 */
extern code u8 DS1302_Read_adr[7];

/**
 * DS1302时间数据数组（BCD编码）

 */
extern u8 DS1302_Time_Reset[7];

/**
 * 十进制时间数据数组

 */

//============================
// 函数声明
//============================

/**
 * @brief DS1302批量写入时间函数
 * @note  调用前需设置DS1302_Time_Reset数组的值
 */
void DS1302_Write_Time(void);

/**
 * @brief DS1302批量读取时间函数
 * @note  读取结果存入Time10数组（十进制格式）
 */
void DS1302_Read_Time(void);

#endif /* __DS1302_H__ */

## 三、DS1302.c 驱动文件编写

### 步骤1：创建文件
**文件路径**: `Hardware/DS1302.c`

### 步骤2：写入内容

```c
/*
	# DS1302实时时钟驱动代码
	1. 本驱动配合蓝桥杯CT107D开发板使用
	2. 底层通信函数（Write_Ds1302_Byte/Read_Ds1302_Byte）由赛题BSP提供
	3. 本文件实现：数组定义 + 读写函数封装
*/

#include <STC15F2K60S2.H>
#include "intrins.h"
#include "Public.h"         // 包含u8/u16/u32等统一类型定义
#include "DS1302.h"         // 自身头文件

//============================
// 寄存器地址数组（code存储在Flash，不占用RAM）
//============================

/**
 * DS1302时间寄存器写地址数组
 * 使用code关键字将数组存储在程序存储区（Flash），节省RAM空间
 * 顺序：秒、分、时、日、月、星期、年
 *
 * DS1302寄存器规则：
 * - 写地址为偶数（bit0=0）：0x80, 0x82, 0x84, 0x86, 0x88, 0x8A, 0x8C
 * - 读地址为奇数（bit0=1）：0x81, 0x83, 0x85, 0x87, 0x89, 0x8B, 0x8D
 * - bit7必须置1（DS1302命令字节最高位固定为1）
 */
code u8 DS1302_Write_adr[7] = {0x80, 0x82, 0x84, 0x86, 0x88, 0x8a, 0x8c};
code u8 DS1302_Read_adr[7]  = {0x81, 0x83, 0x85, 0x87, 0x89, 0x8b, 0x8d};

//============================
// 时间数据数组
//============================

/**
 * DS1302时间数据数组（BCD编码）
 * 用于写入或从DS1302读取的原始时间数据
 * 顺序：秒[0]、分[1]、时[2]、日[3]、月[4]、星期[5]、年[6]
 *
 * 示例初始值：0x55(秒), 0x05(分), 0x23(时), 0x09(日), 0x02(月), 0x07(星期), 0x25(年)
 * 表示：2025年2月9日 23:05:55 星期日
 *
 * BCD编码说明：高4位=十位，低4位=个位
 * 例如0x55 = 十位5，个位5，表示55
 */
u8 DS1302_Time_Reset[7] = {0x55, 0x05, 0x23, 0x09, 0x02, 0x07, 0x25};

/**
 * 十进制时间数据数组
 * 通过BCD转换后得到的十进制时间值
 * 顺序：秒[0]、分[1]、时[2]、日[3]、月[4]、星期[5]、年[6]
 */
u8 Time10[7] = {0, 0, 0, 0, 0, 0, 0};

/**
 * 全局循环变量
 * 定义在全局避免重复在栈上分配，提高执行效率
 */
u8 i = 0;

//============================
// 引脚定义（CT107D开发板连接）
//============================

/**
 * DS1302引脚位定义
 * - SCK (P1^7)：串行时钟线，上升沿写入，下降沿读取
 * - SDA (P2^3)：串行数据线，双向IO
 * - RST (P1^3)：复位/使能线（对应DS1302 datasheet中的CE，高电平有效）
 *
 * 通信时序：RST拉高启动 -> 发送地址 -> 读写数据 -> RST拉低结束
 */
sbit SCK = P1 ^ 7;   // 时钟线（对应DS1302的SCLK引脚）
sbit SDA = P2 ^ 3;   // 数据线（对应DS1302的IO引脚）
sbit RST = P1 ^ 3;   // 复位/使能线（对应DS1302的CE引脚）

//============================
// 函数实现
//============================

/**
 * @brief DS1302批量写入时间函数
 *
 * 功能：将DS1302_Time_Reset数组中的7字节BCD时间数据批量写入DS1302
 *
 * 操作流程：
 * 1. 【关写保护】向0x8E写入0x00（WP=0，允许写入）
 * 2. 【写时间】循环7次，依次写入秒、分、时、日、月、星期、年
 * 3. 【开写保护】向0x8E写入0x80（WP=1，防止误修改）
 *
 * 调用前准备：
 * - 设置DS1302_Time_Reset[0~6]为目标BCD时间值
 * - 例如设置23:30:00：
 *   DS1302_Time_Reset[0] = 0x00;  // 秒
 *   DS1302_Time_Reset[1] = 0x30;  // 分
 *   DS1302_Time_Reset[2] = 0x23;  // 时
 */
void DS1302_Write_Time(void)
{
	Write_Ds1302_Byte(0x8e, 0x00);      // 步骤1：关闭写保护（WP=0）
	
	for (i = 0; i < 7; i++)			    // 步骤2：循环写入7个时间寄存器
	{
		Write_Ds1302_Byte(DS1302_Write_adr[i], DS1302_Time_Reset[i]);
	}
	
	Write_Ds1302_Byte(0x8e, 0x80);      // 步骤3：开启写保护（WP=1）
}

/**
 * @brief DS1302批量读取时间函数
 *
 * 功能：从DS1302读取7字节时间数据，并转换为十进制存入Time10数组
 *
 * 操作流程：
 * 1. 【读BCD数据】循环读取7个寄存器到DS1302_Time_Reset[]（BCD编码）
 * 2. 【BCD转十进制】转换后存入Time10[]
 *
 * BCD转十进制算法：
 * - BCD编码规则：高4位=十位，低4位=个位
 * - 例如：0x35（BCD）= 3×10 + 5 = 35（十进制）
 * - 转换公式：Time10[i] = (DS1302_Time_Reset[i] / 16) * 10 + (DS1302_Time_Reset[i] % 16)
 *
 * 数据分布：
 * - DS1302_Time_Reset[0~6]：原始BCD数据（秒、分、时、日、月、星期、年）
 * - Time10[0~6]：十进制数据（秒、分、时、日、月、星期、年）
 */
void DS1302_Read_Time(void)
{
	// 步骤1：批量读取7个寄存器的BCD时间数据
	for (i = 0; i < 7; i++)
	{
		DS1302_Time_Reset[i] = Read_Ds1302_Byte(DS1302_Read_adr[i]);
	}

	// 步骤2：BCD转十进制
	// 数学推导：
	// 0x35 = 0011 0101（二进制）
	// 高4位 0011 = 3（十位），低4位 0101 = 5（个位）
	// 3×10 + 5 = 35
	for (i = 0; i < 7; i++)
	{
		Time10[i] = (DS1302_Time_Reset[i] / 16) * 10 + (DS1302_Time_Reset[i] % 16);
	}
}
```

## 四、User/main.c 中调用示例

```c
#include "System/Init.h"        // 系统初始化
#include "Hardware/DS1302.h"    // DS1302驱动头文件
#include "Hardware/Nixie.h"     // 数码管驱动（用于显示时间）

void main(void)
{
    // 1. 系统初始化
    System_Init();
    
    // 2. 设置初始时间（只需执行一次，DS1302有电池保持）
    // 设置时间为：2025年2月9日 23:05:55
    DS1302_Time_Reset[0] = 0x55;    // 秒：55
    DS1302_Time_Reset[1] = 0x05;    // 分：05
    DS1302_Time_Reset[2] = 0x23;    // 时：23
    DS1302_Time_Reset[3] = 0x09;    // 日：9号
    DS1302_Time_Reset[4] = 0x02;    // 月：2月
    DS1302_Time_Reset[5] = 0x07;    // 星期：星期日
    DS1302_Time_Reset[6] = 0x25;    // 年：2025
    
    DS1302_Write_Time();            // 写入DS1302芯片
    
    // 3. 主循环：循环读取并显示时间
    while(1)
    {
        DS1302_Read_Time();         // 读取当前时间到Time10数组
        
        // 显示时间（示例：显示时分秒）
        // Time10[2] = 时, Time10[1] = 分, Time10[0] = 秒
        Nixie_DisplayTime(Time10[2], Time10[1], Time10[0]);
    }
}
```

## 五、关键要点速查

| 要点 | 说明 |
|------|------|
| **写地址** | 0x80(秒), 0x82(分), 0x84(时), 0x86(日), 0x88(月), 0x8A(星期), 0x8C(年) |
| **读地址** | 0x81, 0x83, 0x85, 0x87, 0x89, 0x8B, 0x8D（写地址+1） |
| **写保护** | 写入前必须关闭WP（0x8E=0x00），写完开启WP（0x8E=0x80） |
| **BCD编码** | 高4位=十位，低4位=个位。例如0x23 = 23 |
| **时间数组顺序** | [0]秒、[1]分、[2]时、[3]日、[4]月、[5]星期、[6]年 |

## 六、常见问题

**Q1: 时间写入后不生效？**
- 检查是否关闭了写保护（Write_Ds1302_Byte(0x8e, 0x00)）
- 检查BCD编码是否正确（0x09不是0x9）

**Q2: 读取的时间不正确？**
- 检查是否进行了BCD转十进制转换
- 检查读地址是否正确（读地址=写地址+1）

**Q3: 编译报错找不到Write_Ds1302_Byte？**
- 确认BSP文件夹中有ds1302.c/h
- 确认工程已添加BSP文件到编译
