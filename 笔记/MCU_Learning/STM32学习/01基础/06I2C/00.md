#include "stm32f10x.h"
#include "Delay.h"

/* 超时时间定义（防止死循环） */
#define I2C_TIMEOUT         100000     // 超时计数
#define I2C_FLAG_TIMEOUT    -1          // 超时错误码
#define I2C_ERROR          -2          // 通用错误

/* 函数声明 */
void DZ_I2C_Init(void);
int I2C_SendBytes(I2C_TypeDef *I2Cx, uint8_t Addr, uint8_t *pData, uint16_t Size);
int I2C_ReceiveBytes(I2C_TypeDef *I2Cx, uint8_t Addr, uint8_t *pBuffer, uint16_t Size);

int main()

	DZ_I2C_Init();
	
	uint8_t commads[] = {0x00, 0x8D, 0x14, 0xAF, 0xA5};
	
	I2C_SendBytes(I2C1, 0x78, commads, 5);
    while(1)

void DZ_I2C_Init(void)
{
	/* 开启AFIO和GPIOB时钟 */
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_AFIO, ENABLE);
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOB, ENABLE);
	
	/* 配置GPIO引脚（重映射后使用PB8/PB9） */
	GPIO_InitTypeDef GPIO_InitStruct;
	GPIO_InitStruct.GPIO_Pin = GPIO_Pin_8 | GPIO_Pin_9;
	GPIO_InitStruct.GPIO_Mode = GPIO_Mode_AF_OD;
	GPIO_InitStruct.GPIO_Speed = GPIO_Speed_50MHz;
	GPIO_Init(GPIOB, &GPIO_InitStruct);
	
	/* 配置AFIO重映射 */
	GPIO_PinRemapConfig(GPIO_Remap_I2C1, ENABLE);
	
	/* 使能并复位I2C1外设 */
	RCC_APB1PeriphClockCmd(RCC_APB1Periph_I2C1, ENABLE);
	RCC_APB1PeriphResetCmd(RCC_APB1Periph_I2C1, ENABLE);
	RCC_APB1PeriphResetCmd(RCC_APB1Periph_I2C1, DISABLE);
	
	/* 配置I2C工作参数 */
	I2C_InitTypeDef I2C_InitStruct;
	I2C_InitStruct.I2C_ClockSpeed = 400000;
	I2C_InitStruct.I2C_Mode = I2C_Mode_I2C;
	I2C_InitStruct.I2C_DutyCycle = I2C_DutyCycle_2;
	I2C_InitStruct.I2C_OwnAddress1 = 0x00;
	I2C_InitStruct.I2C_Ack = I2C_Ack_Enable;
	I2C_InitStruct.I2C_AcknowledgedAddress = I2C_AcknowledgedAddress_7bit;
	I2C_Init(I2C1, &I2C_InitStruct);
	
	/* 使能I2C */
	I2C_Cmd(I2C1, ENABLE);
}

/**
 * @brief  I2C发送多字节数据（严格遵循I2C协议时序）
 * @param  I2Cx: I2C外设（I2C1/I2C2）
 * @param  Addr: 从机地址（7位，左对齐，bit0=0表示写）
 * @param  pData: 待发送数据缓冲区指针
 * @param  Size: 数据字节数
 * @retval 0: 成功；负数: 错误码
 * 
 * 时序：START -> 地址+W -> ACK -> 数据1 -> ACK -> ... -> 数据N -> ACK -> STOP
 */
int I2C_SendBytes(I2C_TypeDef *I2Cx, uint8_t Addr, uint8_t *pData, uint16_t Size)
{
	uint32_t timeout = I2C_TIMEOUT;
	uint16_t i;
	
	/* 等待总线空闲（总线忙时不能发起通信） */
	while(I2C_GetFlagStatus(I2Cx, I2C_FLAG_BUSY) != RESET)
	{
		if((timeout--) == 0) return I2C_FLAG_TIMEOUT;
	}
	
	/* 产生起始条件（START） */
	I2C_GenerateSTART(I2Cx, ENABLE);
	
	/* 等待起始条件发送完成（EV5: SB=1） */
	timeout = I2C_TIMEOUT;
	while(I2C_CheckEvent(I2Cx, I2C_EVENT_MASTER_MODE_SELECT) != SUCCESS)
	{
		if((timeout--) == 0)
		{
			I2C_GenerateSTOP(I2Cx, ENABLE);  // 超时退出前发停止条件释放总线
			return I2C_FLAG_TIMEOUT;
		}
	}
	
	/* 发送从机地址（写方向，bit0=0） */
	I2C_Send7bitAddress(I2Cx, Addr, I2C_Direction_Transmitter);
	
	/* 等待地址发送完成并收到ACK（EV6: ADDR=1, 方向=发送） */
	timeout = I2C_TIMEOUT;
	while(I2C_CheckEvent(I2Cx, I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED) != SUCCESS)
	{
		if((timeout--) == 0)
		{
			I2C_GenerateSTOP(I2Cx, ENABLE);
			return I2C_FLAG_TIMEOUT;
		}
	}
	
	/* 循环发送数据字节 */
	for(i = 0; i < Size; i++)
	{
		/* 等待发送缓冲区空（EV8: TxE=1） */
		timeout = I2C_TIMEOUT;
		while(I2C_GetFlagStatus(I2Cx, I2C_FLAG_TXE) == RESET)
		{
			if((timeout--) == 0)
			{
				I2C_GenerateSTOP(I2Cx, ENABLE);
				return I2C_FLAG_TIMEOUT;
			}
			
			/* 检查是否收到NACK（从机拒绝接收） */
			if(I2C_GetFlagStatus(I2Cx, I2C_FLAG_AF) != RESET)
			{
				I2C_ClearFlag(I2Cx, I2C_FLAG_AF);  // 清除应答失败标志
				I2C_GenerateSTOP(I2Cx, ENABLE);
				return I2C_ERROR;  // 从机无响应
			}
		}
		
		/* 发送数据字节 */
		I2C_SendData(I2Cx, pData[i]);
	}
	
	/* 等待最后一个字节发送完成（EV8_2: TxE=1, BTF=1） */
	timeout = I2C_TIMEOUT;
	while(I2C_GetFlagStatus(I2Cx, I2C_FLAG_TXE) == RESET || 
		  I2C_GetFlagStatus(I2Cx, I2C_FLAG_BTF) == RESET)
	{
		if((timeout--) == 0)
		{
			I2C_GenerateSTOP(I2Cx, ENABLE);
			return I2C_FLAG_TIMEOUT;
		}
	}
	
	/* 产生停止条件（STOP）释放总线 */
	I2C_GenerateSTOP(I2Cx, ENABLE);
	
	return 0;  // 发送成功
}

/**
 * @brief  I2C接收多字节数据（严格遵循I2C协议时序）
 * @param  I2Cx: I2C外设（I2C1/I2C2）
 * @param  Addr: 从机地址（7位，左对齐，bit0=1表示读）
 * @param  pBuffer: 接收数据缓冲区指针
 * @param  Size: 要接收的字节数
 * @retval 0: 成功；负数: 错误码
 * 
 * 时序：START -> 地址+R -> ACK -> 接收数据1 -> ACK -> ... -> 接收数据N -> NACK -> STOP
 * 注意：最后一个字节前必须关闭ACK，接收最后一个字节后发NACK和STOP
 */
int I2C_ReceiveBytes(I2C_TypeDef *I2Cx, uint8_t Addr, uint8_t *pBuffer, uint16_t Size)
{
	uint32_t timeout = I2C_TIMEOUT;
	uint16_t i;
	
	/* 参数检查 */
	if(Size == 0) return 0;  // 无需接收
	
	/* 等待总线空闲 */
	while(I2C_GetFlagStatus(I2Cx, I2C_FLAG_BUSY) != RESET)
	{
		if((timeout--) == 0) return I2C_FLAG_TIMEOUT;
	}
	
	/* 使能ACK（接收多个字节时，每接收一个字节要回ACK） */
	I2C_AcknowledgeConfig(I2Cx, ENABLE);
	
	/* 产生起始条件（START） */
	I2C_GenerateSTART(I2Cx, ENABLE);
	
	/* 等待起始条件发送完成（EV5: SB=1） */
	timeout = I2C_TIMEOUT;
	while(I2C_CheckEvent(I2Cx, I2C_EVENT_MASTER_MODE_SELECT) != SUCCESS)
	{
		if((timeout--) == 0)
		{
			I2C_GenerateSTOP(I2Cx, ENABLE);
			return I2C_FLAG_TIMEOUT;
		}
	}
	
	/* 发送从机地址（读方向，bit0=1） */
	I2C_Send7bitAddress(I2Cx, Addr, I2C_Direction_Receiver);
	
	/* 等待地址发送完成并收到ACK（EV6: ADDR=1, 方向=接收） */
	timeout = I2C_TIMEOUT;
	while(I2C_CheckEvent(I2Cx, I2C_EVENT_MASTER_RECEIVER_MODE_SELECTED) != SUCCESS)
	{
		if((timeout--) == 0)
		{
			I2C_GenerateSTOP(I2Cx, ENABLE);
			return I2C_FLAG_TIMEOUT;
		}
	}
	
	/* 循环接收数据 */
	for(i = 0; i < Size; i++)
	{
		/* 如果是最后一个字节，提前关闭ACK并准备STOP */
		if(i == (Size - 1))
		{
			/* 关闭ACK，表示最后一个字节后不回ACK（发NACK） */
			I2C_AcknowledgeConfig(I2Cx, DISABLE);
			
			/* 提前请求STOP条件（ hardware会在接收到最后一个字节后自动发STOP） */
			I2C_GenerateSTOP(I2Cx, ENABLE);
		}
		
		/* 等待接收缓冲区非空（EV7: RxNE=1） */
		timeout = I2C_TIMEOUT;
		while(I2C_GetFlagStatus(I2Cx, I2C_FLAG_RXNE) == RESET)
		{
			if((timeout--) == 0)
			{
				I2C_GenerateSTOP(I2Cx, ENABLE);
				return I2C_FLAG_TIMEOUT;
			}
		}
		
		/* 读取数据字节（读取会自动清除RxNE标志） */
		pBuffer[i] = I2C_ReceiveData(I2Cx);
	}
	
	/* 重新使能ACK（为下一次通信做准备） */
	I2C_AcknowledgeConfig(I2Cx, ENABLE);
	
	return 0;  // 接收成功
}
